#define SYS_CLOCK    168
#define SYSTICK_LOAD (168000 - 1)

volatile uint32_t time_tick = 0;

void Time_Update()
{
	time_tick++;
	if (time_tick == 0x400000UL) time_tick = 0;

}


uint32_t millis()
{
	return time_tick;
}


uint32_t micros()
{
		uint32_t SYS_CLOCK_ = SYS_CLOCK;
		uint32_t SYSTICK_LOAD_ = SYSTICK_LOAD;
//	uint32_t SYS_CLOCK_ = sys_clock / 1000000;
//	uint32_t SYSTICK_LOAD_ = sys_clock * 1000 - 1;

	return (time_tick & 0x3FFFFF) * 1000 + (SYSTICK_LOAD_ - SysTick->VAL) / SYS_CLOCK_;
}


uint32_t GetTick()
{
	return time_tick;
}


void delay_ms(uint32_t ms)
{
	uint32_t tickstart = GetTick();

	while ((GetTick() - tickstart) < ms);
}


void delay_us(uint32_t us)
{
	uint32_t temp = micros();
	uint32_t comp = temp + us;
	char flag = 0;
	while (comp > temp)
	{
		if (((time_tick & 0x3FFFFF) == 0) && (flag == 0))
		{
			flag = 1;
		}
		if (flag) temp = micros() + 0x400000UL * 1000;
		else     temp = micros();
	}
}
