void Delay(unsigned int nTime)
{ 
  TimingDelay = nTime;

  while(TimingDelay != 0);
}

{ 
  TimingDelay = nTime/10;

  while(TimingDelay != 0);
}

  delay_us(ms*1000);
}


void TimingDelay_Decrement(void)
{
  if (TimingDelay != 0x00)
  { 
    TimingDelay--;
  }
}


void static_led_pwm()
{
	if(count_pwm == 100)
	{
		count_pwm = 0;
		GPIOA->BSRR = 0x01 << 5;
	}
	else if(count_pwm == pwm_duty)
	{
		GPIOA->BRR = 0x01 << 5;
	}
	
	count_pwm++;
}

	}
	else if(count_servo == servo_duty)
	{
		GPIOA->BRR = 0x01 << 8;
	}
	
	count_servo++;
}

char flag_exit_0 = 0;

void EXTI0_IRQHandler(void)
{
	if (EXTI->PR & (1<<0)) {                       // EXTI0 interrupt pending?
		if(flag_exit_0 == 0)
		{
			flag_exit_0 = 1;
			
			GPIOA->ODR |= (0x01 << 1);
		}
		else
		{
			flag_exit_0 = 0;
			
			GPIOA->ODR  &= ~(0x01 << 1);
		}
		
		EXTI->PR = (1<<0);                          // clear pending interrupt
	}
}

void EXTI3_IRQHandler(void)
{
	int i;
	
	if (EXTI->PR & (1<<3)) {                       // EXTI0 interrupt pending?
		//printf("haha:%d\r\n", rem_count1++);
		
		falling_edge_timing[falling_edge_index] = micros_10us() * 10;
		
		if(falling_edge_index > 0)
		{
		
			diff_falling_edge_timing[falling_edge_index - 1] = 
					falling_edge_timing[falling_edge_index] - falling_edge_timing[falling_edge_index-1];
			
			diff_time1 = diff_falling_edge_timing[falling_edge_index - 1];
			
			if(diff_time > 13000 && diff_time < 14000)
			{
				//printf("Lead Code\r\n");
				
				falling_edge_index = 1;
			}
			else if(diff_time1 > 11000 && diff_time1 < 12000)
			{
				printf("%d..\r\n", remocon_num);
				
				falling_edge_index = -1;
			}
		}
		
		falling_edge_index++;
		if(falling_edge_index == 34)
		{
				exti_callback_flag = 0;
				exti_callback_flag = 1;			
		}
		
		EXTI->PR = (1<<3);                          // clear pending interrupt
	}
}

void rs_high()
{
	GPIOD->ODR |= (0x01 << 11);
}


void rs_low()
{
	GPIOD->ODR &= ~(0x01 << 11);
}


void enbale_high_low()
{
	GPIOD->ODR |= (0x01 << 4);
	GPIOD->ODR &= ~(0x01 << 4);
}

void set_value_8bit(unsigned char value)
{
		GPIOE->ODR &= ~(0xff << 8);
		GPIOE->ODR |= (value << 8);
}


void set_instruction_8bit(unsigned char inst)
{
	rs_low();
	set_value_8bit(inst);
	enbale_high_low();
	rs_high();
	Delay(5);	 // 50 us delay
}

void set_data_8bit(unsigned char data)
{
	set_value_8bit(data);
	enbale_high_low();
	Delay(5);	 // 50 us delay
}

void set_cursor(char row, char col)
{
	if(row == 1)
	{
		set_instruction_8bit(0x80 + col);
	}
	else if(row == 2)
	{
		set_instruction_8bit(0xC0 + col);
	}	
		
}

void charLCD_init()
{
	//backlight_on();
	
	rs_high();
	rw_low();
	
	set_instruction_8bit(0x30);
	set_instruction_8bit(0x30);
	set_instruction_8bit(0x30);
	set_instruction_8bit(0x38);
	set_instruction_8bit(0x06);
	set_instruction_8bit(0x0c);	
	set_instruction_8bit(0x80);
	set_instruction_8bit(0x01);
	Delay(200);  // 2ms delay
}

void charLCD_sting(char *string)
{
	int i;
	
	for(i=0;i<strlen(string);i++)
	{
		set_data_8bit(string[i]);
	}
	
}

volatile unsigned int millis_value = 0;

unsigned long micros_()
{
	return ( ((unsigned long)millis_value)*1000 + TIM2->CNT);
}

unsigned int millis_()
{
	return millis_value;
}

void TIM2_IRQHandler(void) 
{
	 if ((TIM2->SR & 0x0001) != 0) 
	 {
		millis_value++;

		TIM2->SR &= ~(1<<0);
	 }
}

int pwm_count_for_timer = 20;
void TIM3_IRQHandler (void) {

	if((TIM3->SR & 0x0001) != 0) {  // Update Interrupt : 2 ms 마다
		GPIOA->ODR |= (0x01 << 5);

		TIM3->SR &= ~(1<<0);
	}	

	if((TIM3->SR & 0x0002) != 0) {  // Capture Compare : 0~100단계(1단계:20) - 0 ~ 2000
		GPIOA->ODR &= ~(0x01 << 5);

		TIM3->SR &= ~(1<<1);
	}		 
} 

int servo_count_for_timer = 700;
void TIM4_IRQHandler (void) {

	if((TIM4->SR & 0x0001) != 0) {  // Update Interrupt : 20 ms 마다
		GPIOA->ODR |= (0x01 << 9);

		TIM4->SR &= ~(1<<0);
	}	

	if((TIM4->SR & 0x0002) != 0) {  // Capture Compare : 
		GPIOA->ODR &= ~(0x01 << 9);

		TIM4->SR &= ~(1<<1);
	}		 
}
