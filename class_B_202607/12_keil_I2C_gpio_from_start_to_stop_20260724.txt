 #define SCL	0
 #define SDA 	1

void eeprom_byte_write(uint8_t data)
{
	int i = 0;

	for(i=0;i<8;i++)
	{
		// (0x01 << i) : 0 0 0 0_0 0 0 1 << 0 ==> 0 0 0 0_0 0 0 1 &
		//				 0 0 0 0_0 0 0 1 << 1 ==> 0 0 0 0_0 0 1 0 &
		//				 0 0 0 0_0 0 0 1 << 2 ==> 0 0 0 0_0 1 0 0 &
		//										  0 0 0 0_0 0 0 1
		if( (data & (0x01 << (7-i)) ) > 0)
		{
			GPIOA->ODR |= (0x01 << SDA);  // 0 : SDA Set
		}
		else
		{
			GPIOA->ODR &= ~(0x01 << SDA);  // 0 : SDA Clear
		}

		// (0x01 << SCL) : 0x01 을 SCL(0) 만큼 좌쉬프트한다. => ( 0 0 0 0_0 0 0 1 << 0 ) ==> 0x01 값 자체이다.
		// ( GPIOA->ODR | 0x01 ) 한 값을 GPIOA->ODR 에 대입한다.(GPIOA->ODR = GPIOA->ODR | 0x01)
		// 복합대입연산자 를 이용해서 정리하면  [GPIOA->ODR |= (0x01 << SCL) ] 된다.
		GPIOA->ODR |= (0x01 << SCL); // SCL High
		Delay_ms(1);
		GPIOA->ODR &= ~(0x01 << SCL); // SCL Low
		Delay_ms(1);
	}
}



void loop()
{
	int i = 0;
	uint8_t eeprom_addr = 0x01;
	uint8_t eeprom_data = 0xaa;

	// while(1)

	//-------------------------------
	// EEPROM I2C 통신
	// I2C Pull Up 초기화 상태
	gpioA_High(SCL);  // 
	Delay_ms(1);	
	gpioA_High(SDA);
	Delay_ms(1);	
	// Start Condition : 
	// SCK 가 High 상태에서 SDA 가 Low 로 바뀌는 상태
	gpioA_Low(SDA);
	Delay_ms(1);	
	// SCL Low : 데이터를 넣기 위해서 SCL 을 Low 로 한 다음에 SCL 을 계속 High->Low 로 한다.
	gpioA_Low(SCL); 
	Delay_ms(1);	
	// Device Address 8 bit :0xA0(Write bit[0] 포함) => 1 0 1 0_0 0 0 0
	gpioA_High(SDA);  // 0 bit : 1
	Delay_ms(1);
	gpioA_High(SCL);  // SCL High
	Delay_ms(1);
	gpioA_Low(SCL);   // SCL Low
	Delay_ms(1);
	//---------
	gpioA_Low(SDA);  // 1 bit : 0
	Delay_ms(1);
	gpioA_High(SCL);  // SCL High
	Delay_ms(1);
	gpioA_Low(SCL);   // SCL Low
	Delay_ms(1);
	//---------
	gpioA_High(SDA);  // 2 bit : 1
	Delay_ms(1);
	gpioA_High(SCL);  // SCL High
	Delay_ms(1);
	gpioA_Low(SCL);   // SCL Low
	Delay_ms(1);
	//---------

	Delay_ms(1);
	gpioA_High(SCL);  // SCL High
	Delay_ms(1);
	gpioA_Low(SCL);   // SCL Low
	Delay_ms(1);
	//--------------------
	// ACK
	gpioA_High(SCL);  // SCL High
	Delay_ms(1);
	gpioA_Low(SCL);   // SCL Low
	Delay_ms(1);
	//---------
	// Stop Condition
	// gpioA_Low(SDA);  //  : 0
	// Delay_ms(1);
	// gpioA_High(SCL);  // SCL High
	// Delay_ms(1);

	//-------------------------------
	// EEPROM Word Address(메모리 주소)

		// (0x01 << i) : 0 0 0 0_0 0 0 1 << 0 ==> 0 0 0 0_0 0 0 1 &
		//				 0 0 0 0_0 0 0 1 << 1 ==> 0 0 0 0_0 0 1 0 &
		//				 0 0 0 0_0 0 0 1 << 2 ==> 0 0 0 0_0 1 0 0 &
		//										  0 0 0 0_0 0 0 1

		// ( GPIOA->ODR | 0x01 ) 한 값을 GPIOA->ODR 에 대입한다.(GPIOA->ODR = GPIOA->ODR | 0x01)
		// 복합대입연산자 를 이용해서 정리하면  [GPIOA->ODR |= (0x01 << SCL) ] 된다.
		GPIOA->ODR |= (0x01 << SCL); // SCL High
		Delay_ms(1);
		GPIOA->ODR &= ~(0x01 << SCL); // SCL Low
		Delay_ms(1);
	}

	Delay_ms(1);

	//-------------------------------

	eeprom_byte_write(eeprom_data);

volatile void userFn10(void)
{
	if( digitalReadBit(GPIO_C, 10) ){
		printf("External Interrupt GPIOC.10. ==> Key Down..\n");
		digitalWriteBit(GPIO_A, 0, HIGH);
	}
	else{
		printf("External Interrupt GPIOC.10. ==> Key Up..\n");
		digitalWriteBit(GPIO_A, 0, LOW);
	}
}

// user function to be serviced when timer2  interrupt occur
// called in TIM2_IRQ Handler
volatile void userFnTim2(void)
{
	state0 = !state0;
	if(state0)
		digitalWriteBit(GPIO_B, 0, HIGH);
	else
		digitalWriteBit(GPIO_B, 0, LOW);
}

// user function to be serviced when timer3  interrupt occur
// called in TIM3_IRQ Handler
volatile void userFnTim3(void)
{
	state1 = !state1;
	if(state1)
		digitalWriteBit(GPIO_B, 1, HIGH);
	else
		digitalWriteBit(GPIO_B, 1, LOW);
}
